## 1. 概述

本文档详细描述 ELEVEN Blog Tuner 系统中文章生成功能的完整逻辑和流程实现，包括两种生成模式：

1. **笔记整合模式** (默认) - 基于笔记内容检索和整合生成文章
2. **风格+主题模式** - 基于风格配置和主题生成文章

**混合检索** (`_hybrid_retrieval`):
```python
async def _hybrid_retrieval(topic, top_k, user_id):
    # 1. 并行执行两种检索
    vector_result, keyword_result = await asyncio.gather(
        _vector_retrieval(topic, top_k//2, user_id),
        _keyword_retrieval(topic, top_k//2, user_id)
    )
    
    # 2. 合并结果并去重
    all_chunks = {}
    
    # 向量结果加权 (1.2x)
    for chunk in vector_result.chunks:
        key = f"{chunk.note_id}_{hash(chunk.content[:100])}"
        all_chunks[key] = NoteChunk(..., score=chunk.score * 1.2)
    
    # 关键词结果加权 (1.0x)
    for chunk in keyword_result.chunks:
        key = f"{chunk.note_id}_{hash(chunk.content[:100])}"
        if key in all_chunks:
            # 已存在则提升分数
            all_chunks[key].score = max(existing.score, chunk.score)
        else:
            all_chunks[key] = NoteChunk(..., score=chunk.score * 1.0)
    
    # 3. 排序并返回
    chunks = sorted(all_chunks.values(), key=lambda x: x.score, reverse=True)[:top_k]
    return RetrievalResult(chunks, total_notes, strategy)
```

#### 3.1.3 内容格式化

```python
def format_for_prompt(self, retrieval_result):
    """将检索结果格式化为Prompt可用的参考文本"""
    
    sections = []
    sections.append(f"# 参考笔记内容（共{retrieval_result.total_notes}篇笔记）\n")
    
    for i, chunk in enumerate(retrieval_result.chunks, 1):
        sections.append(f"## 参考片段 {i}（来自《{chunk.note_title}》）\n")
        sections.append(chunk.content)
        sections.append("\n")
    
    return "\n".join(sections)
```

```python
async def _generate_by_note_integration(request):
    """笔记整合模式生成文章"""
    
    # Step 1: 检索笔记内容
    if request.note_ids:
        # 使用用户指定的笔记
        retrieval_result = await note_retriever.retrieve_by_notes(
            note_ids=request.note_ids,
            topic=request.topic
        )
    else:
        # 自动检索相关笔记
        retrieval_result = await note_retriever.retrieve_by_topic(
            topic=request.topic,
            top_k=request.top_k,
            strategy=request.retrieval_strategy
        )
    
    # Step 2: 格式化笔记内容
    reference_content = note_retriever.format_for_prompt(retrieval_result)
    
    # Step 3: 加载风格配置（可选）
    if request.style_name:
        style_data = await style_manager.load_style(request.style_name)
        style_config = style_data.get("features", {})
    
    # Step 4: 构建生成提示
    generation_prompt = _build_integration_prompt(
        topic=request.topic,
        reference_content=reference_content,
        style_config=style_config,
        target_length=request.target_length
    )
    
    # Step 5: 生成大纲
    outline = await _generate_outline(
        topic=request.topic,
        reference_content=reference_content,
        style_config=style_config
    )
    
    # Step 6: 生成文章
    content = await _generate_content_with_outline(
        outline=outline,
        prompt=generation_prompt,
        style_config=style_config
    )
    
    # Step 7: 返回结果
    return ArticleGenerationResult(
        content=content,
        mode=GenerationMode.NOTE_INTEGRATION,
```

```python
def _build_integration_prompt(topic, reference_content, style_config, target_length):
    """构建笔记整合模式的生成提示"""
    
    style_instruction = ""
    if style_config:
        style_instruction = f"""
## 写作风格要求
{json.dumps(style_config, ensure_ascii=False, indent=2)}

请确保生成的文章符合上述风格特征。
"""
    
    prompt = f"""请根据以下参考笔记内容，撰写一篇关于"{topic}"的文章。

{reference_content}

---

## 写作要求

1. **主题**：{topic}
2. **内容来源**：基于上述参考笔记内容进行整合和扩展
3. **文章长度**：约 {target_length} 字
4. **写作原则**：
   - 充分吸收参考笔记的核心观点和关键信息
   - 重新组织结构，形成连贯的文章
   - 保持逻辑清晰，层次分明
   - 可以适当补充和扩展，但要基于参考内容
{style_instruction}

请直接输出完整的文章内容。
"""
    return prompt
```

```python
  {{"title": "章节标题", "description": "章节内容简介", "word_count": "预计字数"}},

要求：
- 包含引言、主体内容、总结
```

```python
async def _generate_content_with_outline(outline, prompt, style_config):
    """根据大纲生成完整文章内容"""
    
    sections = []
    
    # 生成引言
    intro_prompt = f"{prompt}\n\n请先撰写引言部分。"
    intro = await _generate_section(intro_prompt, style_config)
    sections.append(intro)
    
    # 生成各章节
    for i, section in enumerate(outline):
        section_prompt = f"""{prompt}

正在撰写第 {i+1} 部分：{section['title']}
内容简介：{section.get('description', '')}
预计字数：{section.get('word_count', '适量')}

请撰写这一部分内容："""
        
        section_content = await _generate_section(section_prompt, style_config)
        sections.append(f"\n## {section['title']}\n\n{section_content}")
    
    # 生成总结
    conclusion_prompt = f"{prompt}\n\n最后，请撰写总结部分。"
    conclusion = await _generate_section(conclusion_prompt, style_config)
    sections.append(f"\n## 总结\n\n{conclusion}")
    
    return "\n\n".join(sections)
```

### 4.1 TaskManager 处理文章生成任务

```python
# gateway/task_manager.py

async def _process_article_generation(self, task: Task):
    """处理文章生成任务"""
    task.add_step("开始文章生成")
    task.update_progress(20)
    
    try:
        # 解析输入数据
        import json
        input_data = json.loads(task.input_data) if isinstance(task.input_data, str) else task.input_data
        
        # 获取 Integration 模块
        from eleven_blog_tunner.gateway.integration import Integration
        integration = Integration()
        
        task.add_step("调用 Integration 执行任务链")
        task.update_progress(30)
        
        # 执行任务链
        result = await integration.execute_task_chain(
            task_type="article_generation",
            input_data=task.input_data,
            task_id=task.task_id
        )
        
        if result.get("success"):
            task.update_progress(90)
            task.add_step("文章生成完成")
            
            # 获取文章ID并更新数据库
            article_id = input_data.get("article_id")
            if article_id:
                from eleven_blog_tunner.common.models import get_db_session, Article
                db = get_db_session()
                try:
                    article = db.query(Article).filter(Article.id == article_id).first()
                    if article:
                        article.content = result.get("result", {}).get("content", "")
                        article.status = "draft"
                        article.word_count = len(article.content)
                        article.metadata = article.metadata or {}
                        article.metadata.update({
                            "generated_at": datetime.utcnow().isoformat(),
                            "generation_result": result.get("result", {})
                        })
                        db.commit()
                finally:
                    db.close()
            
            task.set_result(result.get("result"))
        else:
            task.set_error(result.get("error", "生成失败"))
            
    except Exception as e:
        logger.exception(f"文章生成任务处理失败: {e}")
        task.set_error(str(e))
```

```python
# gateway/integration.py

async def execute_task_chain(
    self,
    task_type: str,
    input_data: str,

) -> Dict[str, Any]:
    """
    执行任务链
    
    完整的任务执行流程：
    1. 获取系统配置 (SystemAgent)
    2. 获取风格配置 (SystemAgent)
    3. 调用 BossAgent 协调执行
    """
    try:
        logger.info(f"开始执行任务链: {task_type}, 任务ID: {task_id}")
        
        # 1. 获取系统配置
        system_config = await self.get_system_config()
        logger.debug(f"系统配置: {system_config}")
        
        # 2. 获取风格配置（如果需要）
        style_config = {}
        if task_type in ["article_generation", "style_analysis"]:
            style_config = await self.get_style_config()
            logger.debug(f"风格配置: {style_config}")
        
        # 3. 调用 BossAgent 执行任务
        boss_result = await self.call_boss_agent(
            task_type,
            input_data,
            task_id
        )
        
        if not boss_result.get("success"):
            logger.error(f"BossAgent 执行失败: {boss_result.get('error')}")
            return {
                "success": False,
                "error": boss_result.get("error", "任务执行失败")
            }
        
        # 4. 构建完整结果
        result = {
            "success": True,
            "result": boss_result.get("result"),
            "metadata": {
                "task_type": task_type,
                "system_config": system_config,
                "style_config": style_config,
                "timestamp": datetime.now().isoformat()
            }
        }
        
        logger.info(f"任务链执行完成: {task_id}")
        return result
        
    except Exception as e:
        logger.error(f"执行任务链失败: {e}")
        return {
            "success": False,
            "error": str(e)
        }
```

```python
async def _execute_note_integration(self, context: AgentContext) -> Dict[str, Any]:
    """

    1. 解析输入参数
    2. 调用笔记检索服务
    3. 调用文章生成服务
```

```python
@router.post("/generate-v2")
async def generate_article_v2(request: ArticleGenerateRequestV2, db: Session):
    """
    文章生成 V2 接口（经过 Gateway 层）
    
    完整流程：
    1. 创建文章记录
    2. 创建 Gateway 任务
    3. TaskManager 调度执行
    4. Integration 协调 Agent
    5. BossAgent 执行生成
    
    支持两种模式：
    1. note_integration - 笔记整合模式（默认）
    2. style_topic - 风格+主题模式
    """
    
    # 1. 创建文章记录
    article_id = uuid.uuid4()
    article = Article(
        id=article_id,
        user_id=uuid.uuid4(),  # TODO: 从认证获取
        title=request.topic,
        status="generating",
        outline=[s.model_dump() for s in request.outline] if request.outline else None,
        word_count=0,
        version=1,
        metadata={
            "target_length": request.target_length,
            "style_name": request.style_name,
            "mode": request.mode.value,
            "note_ids": request.note_ids,
            "retrieval_strategy": request.retrieval_strategy,
            "requested_at": datetime.utcnow().isoformat()
        },
        created_at=datetime.utcnow(),
        updated_at=datetime.utcnow()
    )
    db.add(article)
    db.commit()
    
    # 2. 创建 Gateway 任务
    from eleven_blog_tunner.gateway.task_manager import TaskManager
    task_manager = TaskManager()
    
    task_input = json.dumps({
        "topic": request.topic,
        "mode": request.mode.value,
        "style_name": request.style_name,
        "note_ids": request.note_ids,
        "retrieval_strategy": request.retrieval_strategy,
        "top_k": request.top_k,
        "target_length": request.target_length,
        "outline": [s.model_dump() for s in request.outline] if request.outline else None,
        "article_id": str(article_id),
        "user_id": str(article.user_id)
    })
    
    task_id = await task_manager.create_task(
        task_type="article_generation",
        input_data=task_input,
        user_id=str(article.user_id)
    )
    
    # 3. 启动 TaskManager（如果未启动）
    if not task_manager.running:
        await task_manager.start()
    
    # 4. 返回任务ID
    return SuccessResponse.build(
        data={
            "task_id": str(article_id),
            "gateway_task_id": task_id,
            "status": "generating"
        },
        message=f"文章生成任务已创建（{request.mode.value}模式），主题：{request.topic}"
    )
```

## 7. 错误处理

### 7.1 常见错误场景

| 错误场景 | 处理方式 | 返回信息 |
|---------|---------|---------|
| 未找到相关笔记 | 返回空内容提示 | "未找到与主题相关的笔记内容" |
| 风格不存在 | 使用默认风格继续 | 警告日志，使用默认配置 |
| LLM调用失败 | 重试3次后失败 | 返回错误信息，状态设为failed |
| 大纲解析失败 | 使用默认大纲 | 简单三段式结构 |

### 7.2 错误处理代码示例

```python
try:
    result = await generator.generate(gen_request)
except StyleNotFoundError:
    logger.warning(f"风格不存在: {request.style_name}")
    gen_request.style_name = None

except LLMError as e:
    logger.error(f"LLM调用失败: {e}")
```

## 8. 性能优化

### 8.1 异步处理

- 文章生成使用 `asyncio.create_task()` 在后台执行
- API 立即返回任务ID，不等待生成完成
- 客户端通过轮询 `/articles/{id}/status` 查询进度

### 8.2 检索优化

- 向量检索使用 ChromaDB 持久化存储
- 支持批量向量化
- 混合检索并行执行向量和关键词检索

### 8.3 缓存策略

- 风格配置缓存（避免重复加载）
- 笔记检索结果可缓存（相同主题）
